최종 코드

HTML

<!doctype html>
<html lang="ko">
<head>
  <meta charset="utf-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1" />
  <link rel="stylesheet" href="style.css" />
</head>
<body>
  <div class="slider-wrap">
    <div class="slider">
      <img src="slider/0001.png"/>
      <img src="slider/0002.png"/>
      <!-- ... -->
    </div>
  </div>
  <script src="script.js"></script>
</body>
</html>

CSS

body {
  margin: 0;
  background: black;
  overflow: hidden;
}

.slider-wrap {
  width: 100%;
  overflow: hidden;
}

.slider {
  display: flex;
  transition: transform 0.35s ease-out;
}

.slider img {
  width: 100vw;
  height: 100vh;
  object-fit: contain;
  flex-shrink: 0;
}

JavaScript

/*초기설정*/
const slider = document.querySelector(".slider");
const total = slider.children.length;
let index = 0;

/*슬라이드 이동 함수*/
function moveSlide() {
  if (index === 6) {
    slider.style.transition = "transform 1s ease-out";
  } else {
    slider.style.transition = "transform 0.35s ease-out";
  }
  slider.style.transform = `translateX(-${index * 100}vw)`;
}

/*클릭으로 다음*/
document.addEventListener("click", () => {
  index++;
  if (index >= total) index = 0;
  moveSlide();
});

/*키보드 방향키*/
document.addEventListener("keydown", (e) => {
  if (e.key === "ArrowRight") {
    index++;
    if (index >= total) index = 0;
  }
  if (e.key === "ArrowLeft") {
    index--;
    if (index < 0) index = total - 1;
  }
  moveSlide();
});
